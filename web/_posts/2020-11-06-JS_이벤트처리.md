# 이벤트 처리, 자바 스크립트에서 CSS  

가장 쉬운 방법

```javascript
<!DOCTYPE html>
<html lang="ko" dir="ltr">
  <head>
    <meta charset="utf-8">
    <title></title>
    <style media="screen">
      div.wrapper{
        width : 800px;
        height : 600px;
        background-color: pink;
      }
    </style>
    <script type="text/javascript">
      function change(color){
        var data = document.getElementsByClassName("wrapper");
        data[0].style.backgroundColor = color;



      }
    </script>
  </head>
  <body>
    <div class="wrapper">
      <input type="button" name="" value="red" onclick ="change('red');">
      <input type="button" name="" value="blue" onclick ="change('blue');">
      <input type="button" name="" value="yellow" onclick ="change('yellow');">

    </div>
  </body>
</html>

```

```javascript
<!DOCTYPE html>
<html lang="ko" dir="ltr">
  <head>
    <meta charset="utf-8">
    <title></title>
<style media="screen">
div.wrapper {
   width :  800px;
   height : 600px;
   margin : 0 auto;
   text-align: center;
}
img{
  width: 500px;
  height :  350px;
}

</style>

<script type="text/javascript">
function replace(mypicture){
  var clicked = mypicture.getAttribute("data-img");
  var target = document.getElementById("target")
  target.setAttribute("src",clicked);
}

</script>
  </head>
  <body>
    <div class="wrapper">
      <!-- data-내가 주고싶은 속성 -->
    <a href="#" data-img="dog.jpg" onmouseover="replace(this);">우리 핀이1</a>&nbsp;
    <a href="#" data-img ="dog2.jpg" onmouseover ="replace(this);">우리 핀이2</a>&nbsp;
    <a href="#" data-img ="dog3.jpg" onmouseover ="replace(this);">우리 핀이3</a>&nbsp;
    <a href="#" data-img ="THIS.jpg" onmouseover ="replace(this);">공원</a>&nbsp;<br>
    <img id= "target" src = "dog.jpg">
  </div>


  </body>
</html>

```
